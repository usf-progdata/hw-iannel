view(dictionary) 

#Reverse code items

  filter(Keying == -1) %>% 
  pull(Item)

  mutate(
    across(all_of(reversed),
           ~ 6 - .x,
           .names = "{.col}r")
  )

  rowwise() %>% 
  mutate(
   green_tot = mean(c(green1,green2r,green3,green4,green5), na.rm = FALSE),
